18bool collisionTrianglePoint(Vector2f a, Vector2f b, Vector2f c, Vector2f point) // checks if a point is within the triangle ABC : the point must be at the left of each edge -> be careful to the winding direction
19{
20 Vector2f ab = b - a;

22 Vector2f ca = a - c;

31vector<Triangle> triangulate(vector<Vector2f> points) // the ear clipping algorithm
32{
33 vector<Triangle> triangles; /* a dynamic array that will store the points of the triangles : if the triangle n is (An Bn Cn), then the points will be stored as [A1,B1,C1,
34 A2,B2,C2,
35 A3,B3,C3...] */
36 vector<Vector2f> initialPoints = points;
37 if (points.size() < 3) // let's make sure that the user don't feed the function with less than 3 points !
38 return triangles;
39 else
40 {
41 bool impossibleToTriangulate = false;
42 bool triangleFound = true;
43
44 while (points.size() != 0) // run the algorithm until our polygon is empty
45 {
46 if (!triangleFound) // if we've looped once without finding any ear, the program is stuck, the polygon is not triangulable for our algorithm (likely to be a 8 shape or such self intersecting polygon)
47 return triangles;
48
49 triangleFound = false; // we want to find a new ear at each loop
50
51 for (int i(0); i < points.size() - 2; i++) // for each 3 consecutive points we check if it's an ear : an ear is a triangle that wind in the right direction and that do not contain any other point of the polygon
52 {
53 if (!triangleFound) // if we still didn't find an ear
54 {
55 bool result = false;
56 if (determinant(points[i + 1] - points[i], points[i + 2] - points[i + 1]) > 0) // if the triangle winds in the right direction
57 {
58 result = true;
59 for (int j(0); j < initialPoints.size(); j++) // we check if there's no point inside it
60 {
61 if (collisionTrianglePoint(points[i + 2], points[i + 1], points[i], initialPoints[j]))
62 {
63 result = false; // if I got a point in my triangle, then it's not an ear !
64 }
65 }
66 }
67
68 if (result) // now, we have found an ear :
69 {
70 triangleFound = true;
71
72 triangles.emplace_back(Triangle(points[i], points[i + 1], points[i + 2]));
73
74 vector<Vector2f> bufferArray;
75 for (int j(0); j < points.size(); j++) // then we delete the triangle in the points array : we already know that it's an ear, we don't need it anymore
76 {
77 if (j != i + 1) // we copiy all the points in a buffer array except the point we don't want
78 {
79 bufferArray.push_back(points[j]);
80 }
81 }
82 points = bufferArray;
83 }
84 }
85 }
86 }
87 }
88 return triangles; // we return the triangle array
89}
